<objective>
## Goal
Build the PromptTab client component: loads the current design schema via GET `/api/schemas`, renders all 6 sections with editable fields and per-section lock toggles, tracks changes with a modified indicator, and saves via PATCH `/api/schemas/[id]`. Wire it into the workspace Prompt tab in place of PlaceholderTab.

## Purpose
This is the user-facing control surface for Phase 4. Users need to safely customize extracted design logic (change colors, composition, typography) and lock style-critical sections before Agent B2 generates the final prompt. Without this editor, the schema is read-only and locked_fields is always empty — Phase 5 cannot selectively preserve intent.

## Output
- `src/app/projects/[id]/prompt-tab.tsx` — full client component
- `src/app/projects/[id]/page.tsx` — Prompt tab wired to `<PromptTab>`

## Task 2 — Create `prompt-tab.tsx`

**File:** `src/app/projects/[id]/prompt-tab.tsx`

**Action:** Create the full client component. Follow this exact structure:

### State machine
```ts
type Phase = "loading" | "empty" | "ready" | "saving" | "error"
```

### Types (co-located, same pattern as analyze-tab.tsx)
```ts
import type { DesignExtraction } from "@/lib/schemas/design-extraction"
import type { DesignSchema } from "@/types/db"

interface EditableSchema {
  id: string
  frame: DesignExtraction["frame"] | null
  palette: DesignExtraction["palette"] | null
  layout: DesignExtraction["layout"] | null
  type_scale: DesignExtraction["type_scale"]
  text_fields: DesignExtraction["text_fields"]
  elements: DesignExtraction["elements"]
}
```

### parseDbSchema helper (same parsing logic as analyze-tab, but returns EditableSchema)
```ts
function parseDbSchema(raw: DesignSchema): EditableSchema {
  return {
    id: raw.id,
    frame: raw.frame ? JSON.parse(raw.frame) : null,
    palette: raw.palette ? JSON.parse(raw.palette) : null,
    layout: raw.layout ? JSON.parse(raw.layout) : null,
    type_scale: raw.type_scale ? JSON.parse(raw.type_scale) : null,
    text_fields: raw.text_fields ? JSON.parse(raw.text_fields) : [],
    elements: raw.elements ? JSON.parse(raw.elements) : [],
  }
}
```

### State
```ts
const [phase, setPhase] = useState<Phase>("loading")
const [schema, setSchema] = useState<EditableSchema | null>(null)
const [original, setOriginal] = useState<EditableSchema | null>(null)
const [lockedFields, setLockedFields] = useState<string[]>([])
const [errorMsg, setErrorMsg] = useState("")
```

`isDirty` is derived (not stored):
```ts
const isDirty =
  JSON.stringify(schema) !== JSON.stringify(original) ||
  JSON.stringify(lockedFields) !== JSON.stringify(originalLocked)
```
Store `originalLocked` separately (a second `useState<string[]>([])`).

### On mount: fetch schema
```ts
useEffect(() => {
  void loadSchema()
}, [projectId])

async function loadSchema() {
  setPhase("loading")
  try {
    const res = await fetch(`/api/schemas?projectId=${projectId}`)
    if (res.status === 404) { setPhase("empty"); return }
    if (!res.ok) { setErrorMsg("Failed to load schema."); setPhase("error"); return }
    const raw = await res.json() as DesignSchema
    const parsed = parseDbSchema(raw)
    const locks = JSON.parse(raw.locked_fields) as string[]
    setSchema(parsed)
    setOriginal(parsed)
    setLockedFields(locks)
    setOriginalLocked(locks)
    setPhase("ready")
  } catch {
    setErrorMsg("Failed to load schema. Check the dev server.")
    setPhase("error")
  }
}
```

### Lock toggle helper
```ts
function toggleLock(section: string) {
  setLockedFields(prev =>
    prev.includes(section) ? prev.filter(s => s !== section) : [...prev, section]
  )
}
```

### isModified helper (per-section diff dot)
```ts
function isModified(section: keyof Omit<EditableSchema, "id">): boolean {
  if (!original || !schema) return false
  return JSON.stringify(schema[section]) !== JSON.stringify(original[section])
}
```

### Save function
```ts
async function save() {
  if (!schema) return
  setPhase("saving")
  try {
    const res = await fetch(`/api/schemas/${schema.id}`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        frame: schema.frame,
        palette: schema.palette,
        layout: schema.layout,
        type_scale: schema.type_scale,
        text_fields: schema.text_fields,
        elements: schema.elements,
        locked_fields: lockedFields,
      }),
    })
    if (!res.ok) {
      const data = await res.json() as { error?: string }
      setErrorMsg(data.error ?? "Save failed.")
      setPhase("ready")  // stay on ready so user can retry
      return
    }
    // Reset dirty state
    setOriginal(schema)
    setOriginalLocked(lockedFields)
    setPhase("ready")
  } catch {
    setErrorMsg("Save failed. Check the dev server.")
    setPhase("ready")
  }
}
```

### Section header pattern (reuse for all sections):
```tsx
function SectionHeader({
  title,
  section,
  modified,
  locked,
  onToggleLock,
}: {
  title: string
  section: string
  modified: boolean
  locked: boolean
  onToggleLock: () => void
}) {
  return (
    <div className="flex items-center justify-between pb-2">
      <div className="flex items-center gap-2">
        <span className="text-xs font-semibold uppercase tracking-wider text-muted-foreground">
          {title}
        </span>
        {modified && (
          <span className="inline-block h-1.5 w-1.5 rounded-full bg-amber-500" />
        )}
        {locked && (
          <Badge variant="secondary" className="text-xs">locked</Badge>
        )}
      </div>
      <Button
        variant="ghost"
        size="sm"
        className="h-6 px-2 text-xs"
        onClick={onToggleLock}
      >
        {locked ? "🔒 Unlock" : "🔓 Lock"}
      </Button>
    </div>
  )
}
```
