Request: `GET /api/blueprints?projectId=X`

Steps:
1. Read `projectId` from `searchParams` — return 400 `{ error: "projectId is required" }` if missing
2. Verify project exists — `SELECT id FROM projects WHERE id = ?` — return 404 `{ error: "Project not found" }` if missing
3. Query blueprints — `SELECT id, name, created_at FROM grammar_blueprints WHERE project_id = ? ORDER BY created_at DESC`
4. Return 200 with the array (can be empty)

Response shape: `Array<{ id: string; name: string | null; created_at: string }>`

Pattern to follow: same structure as `GET /api/schemas/route.ts` — `getDb()`, `NextResponse.json()`, try/catch with 500.

Do NOT return full blueprint rows — only the 3 fields needed for the selector. This keeps the payload minimal and avoids exposing raw_prompts to the client unnecessarily.

### Load blueprints on mount

Inside `loadSchema()`, after successfully setting `setPhase("ready")`, add a call to `void loadBlueprints()`. Define `loadBlueprints()` as:

```ts
async function loadBlueprints() {
  try {
    const res = await fetch(`/api/blueprints?projectId=${projectId}`)
    if (!res.ok) return
    const data = (await res.json()) as BlueprintOption[]
    setBlueprints(data)
    if (data.length > 0) {
      setSelectedBlueprintId(data[0].id)  // pre-select most recent (first in DESC order)
    }
  } catch {
    // non-blocking — generation panel will show appropriate state
  }
}
```

### Generate function

```ts
async function generate() {
  setGenerationPhase("generating")
  setPromptText("")
  setOutputId(null)
  setGenerationError("")

  try {
    const res = await fetch("/api/rewrite", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        projectId,
        ...(selectedBlueprintId ? { blueprintId: selectedBlueprintId } : {}),
      }),
    })

    if (!res.ok || !res.body) {
      const data = (await res.json()) as { error?: string }
      setGenerationError(data.error ?? "Generation failed.")
      setGenerationPhase("error")
      return
    }

    // Read X-Output-Id header before consuming body
    const id = res.headers.get("X-Output-Id")
    if (id) setOutputId(id)

    // Stream body as plain text
    const reader = res.body.getReader()
    const decoder = new TextDecoder()
    while (true) {
      const { done, value } = await reader.read()
      if (done) break
      const chunk = decoder.decode(value, { stream: true })
      setPromptText((prev) => prev + chunk)
    }

    setGenerationPhase("complete")
  } catch {
    setGenerationError("Generation failed. Check the dev server.")
    setGenerationPhase("error")
  }
}
```

### Generation panel JSX

Append below the last `</EditorCard>` (Elements card) in the `ready/saving` render, wrapped in a `<Separator />` then a new section:

```tsx
<Separator />

{/* Generation panel */}
<div className="flex flex-col gap-4">
  <div className="flex items-center justify-between">
    <h3 className="text-sm font-semibold">Generate Prompt</h3>
  </div>

  {blueprints.length === 0 ? (
    <div className="rounded-lg border border-dashed p-6 text-center text-sm text-muted-foreground">
      No grammar blueprint found. Run the{" "}
      <span className="font-medium">Blueprint</span> tab first to distil a
      writing style.
    </div>
  ) : (
    <div className="flex flex-col gap-3">
      {/* Blueprint selector */}
      {blueprints.length > 1 && (
        <div className="flex flex-col gap-1">
          <label className="text-xs text-muted-foreground">Blueprint</label>
          <select
            value={selectedBlueprintId ?? ""}
            onChange={(e) => setSelectedBlueprintId(e.target.value)}
            disabled={generationPhase === "generating"}
            className="h-9 rounded-md border border-input bg-background px-3 text-sm disabled:opacity-50"
          >
            {blueprints.map((bp) => (
              <option key={bp.id} value={bp.id}>
                {bp.name ?? `Blueprint (${bp.created_at.slice(0, 10)})`}
              </option>
            ))}
          </select>
        </div>
      )}

      {/* Single blueprint info (no dropdown needed) */}
      {blueprints.length === 1 && (
        <p className="text-xs text-muted-foreground">
          Using:{" "}
          <span className="font-medium text-foreground">
            {blueprints[0].name ?? `Blueprint (${blueprints[0].created_at.slice(0, 10)})`}
          </span>
        </p>
      )}

      <Button
        onClick={() => void generate()}
        disabled={generationPhase === "generating" || !selectedBlueprintId}
        className="self-start"
      >
        {generationPhase === "generating" ? "Generating…" : "Generate"}
      </Button>
    </div>
  )}

  {/* Prompt output display */}
  {(generationPhase === "generating" ||
    generationPhase === "complete" ||
    (generationPhase === "error" && promptText)) && (
    <div className="flex flex-col gap-2">
      <div className="min-h-24 rounded-md border bg-muted/30 p-3 text-sm leading-relaxed whitespace-pre-wrap font-mono">
        {promptText}
        {generationPhase === "generating" && (
          <span className="inline-block h-3 w-1 animate-pulse bg-foreground/60 align-middle" />
        )}
      </div>
      {outputId && (
        <p className="text-xs text-muted-foreground">
          Output ID: <span className="font-mono">{outputId}</span>
        </p>
      )}
    </div>
  )}

  {/* Generation error */}
  {generationPhase === "error" && generationError && (
    <p className="text-xs text-destructive">{generationError}</p>
  )}
</div>
```

Notes:

- The blinking cursor `<span>` uses `animate-pulse` (already in Tailwind base)
- `whitespace-pre-wrap font-mono` for prompt display matches prompt engineering conventions
- The `select` element uses the same border/radius/height tokens as shadcn `Input` — no new component import needed
