    Create `src/lib/env.ts` — typed, validated env config with provider resolution:

    ```typescript
    import { z } from "zod"

    const envSchema = z.object({
      // Mode
      LOCAL_MODE: z.string().transform((v) => v === "true").default("true"),
      IMAGE_GEN_PROVIDER: z.enum(["nano_banana_2", "replicate"]).optional(),

      // Local inference
      OLLAMA_BASE_URL: z.string().url().optional().default("http://localhost:11434"),

      // Cloud AI providers
      OPENAI_API_KEY: z.string().optional(),
      GOOGLE_GENERATIVE_AI_API_KEY: z.string().optional(),
      ANTHROPIC_API_KEY: z.string().optional(),

      // Image generation
      REPLICATE_API_TOKEN: z.string().optional(),

      // Storage (production only)
      NEXT_PUBLIC_SUPABASE_URL: z.string().url().optional(),
      NEXT_PUBLIC_SUPABASE_ANON_KEY: z.string().optional(),
      SUPABASE_SERVICE_ROLE_KEY: z.string().optional(),

      NODE_ENV: z.enum(["development", "production", "test"]).default("development"),
    })

    const parsed = envSchema.safeParse(process.env)

    if (!parsed.success) {
      console.error("❌ Invalid environment variables:", parsed.error.flatten().fieldErrors)
      throw new Error("Invalid environment variables")
    }

    export const env = parsed.data

    /** True when running locally with Ollama inference */
    export const isLocalMode = () => env.LOCAL_MODE === true

    /**
     * Resolves the image generation provider.
     * - If IMAGE_GEN_PROVIDER is explicitly set, honour it.
     * - If LOCAL_MODE=true, default to "replicate" (Ollama handles agents, Replicate handles images locally).
     * - If LOCAL_MODE=false, default to "nano_banana_2" (Gemini API).
     */
    export const getImageGenProvider = (): "nano_banana_2" | "replicate" => {
      if (env.IMAGE_GEN_PROVIDER) return env.IMAGE_GEN_PROVIDER
      return isLocalMode() ? "replicate" : "nano_banana_2"
    }

    /** Resolves the vision/analysis model ID for Agent A */
    export const getVisionModel = () =>
      isLocalMode() ? "qwen3-vl:30b" : "gemini-1.5-flash"

    /** Resolves the text model ID for Agents B1 and B2 */
    export const getTextModel = () =>
      isLocalMode() ? "qwen3-vl:30b" : "gpt-4o-mini"
    ```

    Create `.env.example` — complete reference for all variables:

    ```bash
    # ─── Mode ────────────────────────────────────────────────────────────────────
    # true  → use Ollama for all AI agents; Replicate for image gen
    # false → use Gemini 1.5 Flash (Agent A) + GPT-4o mini (B1/B2) + Nano Banana 2
    LOCAL_MODE=true

    # Override image gen provider regardless of LOCAL_MODE
    # Options: nano_banana_2 | replicate
    # IMAGE_GEN_PROVIDER=nano_banana_2

    # ─── Local Inference (Ollama) ─────────────────────────────────────────────────
    OLLAMA_BASE_URL=http://localhost:11434
    # Required model: qwen3-vl:30b (vision + text)
    # Optional model: cogvlm2:10b (benchmark vs qwen3-vl at Phase 2)

    # ─── Cloud AI Providers ───────────────────────────────────────────────────────
    OPENAI_API_KEY=
    GOOGLE_GENERATIVE_AI_API_KEY=
    ANTHROPIC_API_KEY=

    # ─── Image Generation ─────────────────────────────────────────────────────────
    # Primary (production): Nano Banana 2 uses GOOGLE_GENERATIVE_AI_API_KEY above
    # Fallback (local dev + production override):
    REPLICATE_API_TOKEN=

    # ─── Storage (production only — leave blank for local dev) ───────────────────
    NEXT_PUBLIC_SUPABASE_URL=
    NEXT_PUBLIC_SUPABASE_ANON_KEY=
    SUPABASE_SERVICE_ROLE_KEY=
    ```

    Avoid: Importing env.ts in client components (server-side only); hardcoding model strings anywhere else in the codebase — always import from env.ts.
